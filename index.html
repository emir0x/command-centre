function statusLabel(s){ return s==='todo'?'Not Started':s==='inprogress'?'In Progress':'Done'; }


function deadlineDisplay(dl) {

  if(!dl) return {text:'—',cls:''};

  const d=new Date(dl), today=new Date(); today.setHours(0,0,0,0);

  const diff=Math.round((d-today)/86400000);

  const text=d.toLocaleDateString('en-GB',{day:'numeric',month:'short'});

  if(diff<0) return {text:text+' ⚠',cls:'deadline-overdue'};

  if(diff<=3) return {text,cls:'deadline-soon'};

  return {text,cls:'deadline-ok'};

}


function cycleStatus(type,id){

  const state=getState(), task=state.tasks[type].find(t=>t.id===id);

  if(!task) return;

  const cycle={todo:'inprogress',inprogress:'done',done:'todo'};

  task.status=cycle[task.status]||'todo';

  if(task.status==='done'){task.completedBy=currentUser.id;task.completedAt=Date.now();}

  save(state); renderBoard(type);

}


function deleteTask(type,id){

  if(!confirm('Delete this task?')) return;

  const state=getState();

  state.tasks[type]=state.tasks[type].filter(t=>t.id!==id);

  save(state); renderBoard(type);

}

  grid.innerHTML='';


  for(let i=0;i<21;i++){

    const d=new Date(SPRINT_START); d.setDate(d.getDate()+i);

    const k=d.toISOString().split('T')[0];

    const isToday=k===today, complete=state.streak[k], isPast=d<=new Date();

    const el=document.createElement('div');

    el.className='s-day'+(complete?' complete':'')+(isToday?' today':'');

    el.innerHTML=`<span class="s-day-num">${i+1}</span><span style="font-size:0.42rem">${d.toLocaleDateString('en-GB',{weekday:'short'}).toUpperCase()}</span>`;

    if(isPast&&!isToday) el.onclick=()=>{ const s=getState(); s.streak[k]=!s.streak[k]; save(s); renderStreak(); };

    grid.appendChild(el);

  }


  const ws=document.getElementById('weekStats');

  if(!ws) return;

  ws.innerHTML='';

  for(let w=0;w<3;w++){

    let done=0;

    for(let d=0;d<7;d++){ const dt=new Date(SPRINT_START); dt.setDate(dt.getDate()+w*7+d); if(state.streak[dt.toISOString().split('T')[0]]) done++; }

    const pct=Math.round((done/7)*100);

    ws.innerHTML+=`<div style="margin-bottom:14px"><div style="display:flex;justify-content:space-between;margin-bottom:6px"><span style="font-family:'DM Mono',monospace;font-size:0.58rem;letter-spacing:0.1em;text-transform:uppercase;color:var(--ink3)">Week ${w+1}</span><span style="font-family:'DM Mono',monospace;font-size:0.58rem;color:var(--gold)">${done}/7 days</span></div><div style="height:3px;background:var(--border);border-radius:1px;overflow:hidden"><div style="height:100%;width:${pct}%;background:var(--green);border-radius:1px;transition:width 0.4s"></div></div></div>`;

  }

}
